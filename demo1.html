for (var i = 0; i < 10; i++) {
  // Define a random color
  var red = String(
    Math.floor(Math.random()*250));
  var green = String(
    Math.floor(Math.random()*250));
  var blue = String(
    Math.floor(Math.random()*250));

  // Clone a box
  $('#box').clone().css({
    // Make it visible and position it randomly
    'display': '',
    'top': 
      Math.floor(Math.random()*100) + 20,
    'left': 
      Math.floor(Math.random()*100) + 20,
    'background-color': 
      // Swap comments to use shades of red
      'rgb(' + red + ',' + green + ',' + blue + ')'
      //'rgb(' + red + ',0,0)'
  }).
  // Append to the DOM
  appendTo($('#box_arena')).
  // Animate the box
  animate({
    'top':
      Math.floor(Math.random()*500) + 20,
    'left':
      Math.floor(Math.random()*300) + 20
  }, /* duration */ 500);
}
